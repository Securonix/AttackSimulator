017    package grails.plugins.quartz;
018    
019    import grails.plugins.quartz.config.TriggersConfigBuilder;
020    import grails.util.GrailsUtil;
021    import groovy.lang.Closure;
022    import org.codehaus.groovy.grails.commons.AbstractInjectableGrailsClass;
023    import org.codehaus.groovy.grails.commons.GrailsClassUtils;
024    import org.quartz.JobExecutionContext;
025    
026    import java.util.HashMap;
027    import java.util.Map;
028    
029    import static grails.plugins.quartz.GrailsJobClassConstants.*;
030    
031    
032    /**
033     * Grails artefact class which represents a Quartz job.
034     *
035     * @author Micha?? K??ujszo
036     * @author Marcel Overdijk
037     * @author Sergey Nebolsin (nebolsin@gmail.com)
038     * @since 0.1
039     */
040    public class DefaultGrailsJobClass extends AbstractInjectableGrailsClass implements GrailsJobClass {
041    
042        public static final String JOB = "Job";
043        private Map triggers = new HashMap();
044    
045    
046        public DefaultGrailsJobClass(Class clazz) {
047            super(clazz, JOB);
048            evaluateTriggers();
049        }
050    
051        private void evaluateTriggers() {
052            // registering additional triggersClosure from 'triggersClosure' closure if present
053            Closure triggersClosure = (Closure) GrailsClassUtils.getStaticPropertyValue(getClazz(), "triggers");
054    
055            TriggersConfigBuilder builder = new TriggersConfigBuilder(getFullName());
056    
057            if (triggersClosure != null) {
058                builder.build(triggersClosure);
059                triggers = (Map) builder.getTriggers();
060            } else {
061                // backward compatibility
062                if (isCronExpressionConfigured()) {
063                    GrailsUtil.deprecated("You're using deprecated 'def cronExpression = ...' parameter in the " + getFullName() + ", use 'static triggers = { cron cronExpression: ...} instead.");
064                    triggers = builder.createEmbeddedCronTrigger(getStartDelay(), getCronExpression());
065                } else {
066                    GrailsUtil.deprecated("You're using deprecated 'def startDelay = ...; def timeout = ...' parameters in the" + getFullName() + ", use 'static triggers = { simple startDelay: ..., repeatInterval: ...} instead.");
067                    triggers = builder.createEmbeddedSimpleTrigger(getStartDelay(), getTimeout(), getRepeatCount());
068                }
069            }
070        }
071    
072        public void execute() {
073            getMetaClass().invokeMethod(getReferenceInstance(), EXECUTE, new Object[]{});
074        }
075    
076        public void execute(JobExecutionContext context) {
077            getMetaClass().invokeMethod(getReferenceInstance(), EXECUTE, new Object[]{context});
078        }
079    
080        // TODO: ============== start of deprecated methods =================
081        public long getTimeout() {
082            Object obj = getPropertyValue(TIMEOUT);
083            if (obj == null) return DEFAULT_TIMEOUT;
084            else if (!(obj instanceof Number)) {
085                throw new IllegalArgumentException(
086                        "timeout trigger property in the job class " +
087                                getShortName() + " must be Integer or Long");
088            }
089    
090            return ((Number) obj).longValue();
091        }
092    
093        public long getStartDelay() {
094            Object obj = getPropertyValue(START_DELAY);
095            if (obj == null) return DEFAULT_START_DELAY;
096            else if (!(obj instanceof Number)) {
097                throw new IllegalArgumentException(
098                        "startDelay trigger property in the job class " +
099                                getShortName() + " must be Integer or Long");
100            }
101    
102            return ((Number) obj).longValue();
103        }
104    
105        public int getRepeatCount() {
106            Object obj = getPropertyValue(REPEAT_COUNT);
107            if (obj == null) return DEFAULT_REPEAT_COUNT;
108            else if (!(obj instanceof Number)) {
109                throw new IllegalArgumentException(
110                        "repeatCount trigger property in the job class " +
111                                getShortName() + " must be Integer or Long");
112            }
113    
114            return ((Number) obj).intValue();
115        }
116    
117        public String getCronExpression() {
118            String cronExpression = (String) getPropertyOrStaticPropertyOrFieldValue(CRON_EXPRESSION, String.class);
119            if (cronExpression == null || "".equals(cronExpression)) return DEFAULT_CRON_EXPRESSION;
120            return cronExpression;
121        }
122    
123        public String getGroup() {
124            String group = (String) getPropertyOrStaticPropertyOrFieldValue(GROUP, String.class);
125            if (group == null || "".equals(group)) return DEFAULT_GROUP;
126            return group;
127        }
128    
129        // not certain about this... feels messy
130        public boolean isCronExpressionConfigured() {
131            String cronExpression = (String) getPropertyOrStaticPropertyOrFieldValue(CRON_EXPRESSION, String.class);
132            return cronExpression != null;
133        }
134        // TODO: ============== end of deprecated methods =================
135    
136        public boolean isConcurrent() {
137            Boolean concurrent = (Boolean) getPropertyValue(CONCURRENT, Boolean.class);
138            return concurrent == null ? DEFAULT_CONCURRENT : concurrent;
139        }
140    
141        public boolean isSessionRequired() {
142            Boolean sessionRequired = (Boolean) getPropertyValue(SESSION_REQUIRED, Boolean.class);
143            return sessionRequired == null ? DEFAULT_SESSION_REQUIRED : sessionRequired;
144        }
145    
146        public boolean getDurability() {
147            Boolean durability = (Boolean) getPropertyValue(DURABILITY, Boolean.class);
148            return durability == null ? DEFAULT_DURABILITY : durability;
149        }
150    
151        public boolean getRequestsRecovery() {
152            Boolean requestsRecovery = (Boolean) getPropertyValue(REQUESTS_RECOVERY, Boolean.class);
153            return requestsRecovery == null ? DEFAULT_REQUESTS_RECOVERY : requestsRecovery;
154        }
155    
156        public Map getTriggers() {
157            return triggers;
158        }
159    }
